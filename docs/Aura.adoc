> ## 📁 2. `aura.js`

```javascript
const AURA = (() => {

  // -----------------------------
  // STREAM ENGINE
  // -----------------------------
  const streams = [
    {
      name: "BBC World",
      url: "https://stream.live.vc.bbcmedia.co.uk/bbc_world_service"
    },
    {
      name: "Classic FM",
      url: "https://icecast.thisisglobal.com/classicfm.mp3"
    }
  ];

  let currentIndex = 0;
  const player = () => document.getElementById("player");

  function loadStream(i) {
    const s = streams[i];
    if (!s) return;

    player().src = s.url;
    player().play();

    document.getElementById("status").innerText =
      "Streaming: " + s.name;
  }

  function nextStream() {
    currentIndex = (currentIndex + 1) % streams.length;
    loadStream(currentIndex);
  }

  function renderStreams() {
    const el = document.getElementById("streamList");
    el.innerHTML = streams.map((s, i) => `
      <div class="item" onclick="AURA.selectStream(${i})">
        ${s.name}
        <div class="small">${s.url}</div>
      </div>
    `).join("");
  }

  function selectStream(i) {
    currentIndex = i;
    loadStream(i);
  }

  // -----------------------------
  // PODCAST ENGINE (SIMULATED AI)
  // -----------------------------
  function runPodcast() {
    const mode = document.getElementById("mode").value;
    const prompt = document.getElementById("prompt").value;

    const output = document.getElementById("podcastOutput");

    const script = generateScript(mode, prompt);

    output.innerHTML = `
      <h4>Generated Episode (${mode})</h4>
      <pre>${script}</pre>
    `;

    addToQueue(prompt, mode);
  }

  function generateScript(mode, prompt) {
    if (!prompt) return "No input provided.";

    switch(mode) {
      case "conversation":
        return `
Host A: Today we're discussing ${prompt}.
Host B: Interesting point — let's break it down.
Host A: This connects to broader Web4 systems.
Host B: Exactly, especially in distributed media.
        `;

      case "deepdive":
        return `
Deep Dive Episode: ${prompt}

1. Overview
2. System Breakdown
3. Technical implications
4. Web4 integration layer
        `;

      case "news":
        return `
Breaking Analysis:
Topic: ${prompt}
Summary: AI-generated synthesis of current context.
        `;

      case "debate":
        return `
PRO: ${prompt} enables scalable systems.
CON: But introduces complexity and risk.
        `;
    }
  }

  // -----------------------------
  // INTENT QUEUE (WEB4 MEMORY LITE)
  // -----------------------------
  let queue = [];

  function addToQueue(text, type) {
    queue.push({ text, type, time: Date.now() });
    renderQueue();
  }

  function renderQueue() {
    const el = document.getElementById("queue");
    el.innerHTML = queue.map(q => `
      <div class="item">
        ${q.type.toUpperCase()}: ${q.text}
      </div>
    `).join("");
  }

  // -----------------------------
  // LIVE MODE (SIMULATED SIGNAL LAYER)
  // -----------------------------
  let live = false;

  function toggleLive() {
    live = !live;
    document.getElementById("status").innerText =
      live ? "LIVE MODE ACTIVE" : "Idle";
  }

  // -----------------------------
  // PUBTEETH (EXPORT LAYER)
  // -----------------------------
  function exportEpisode() {
    const blob = new Blob([JSON.stringify(queue, null, 2)], {
      type: "application/json"
    });

    const url = URL.createObjectURL(blob);

    const el = document.getElementById("exports");
    el.innerHTML = `
      <a href="${url}" download="aura_episode.json">
        Download Episode Package
      </a>
    `;
  }

  // -----------------------------
  // INIT
  // -----------------------------
  renderStreams();
  loadStream(0);

  return {
    nextStream,
    selectStream,
    runPodcast,
    generatePodcast: runPodcast,
    toggleLive,
    exportEpisode
  };

})();
```
